31template <typename T, typename TolT = T>
32constexpr bool areNearlyEqual(T val1, T val2, TolT absTol = std::numeric_limits<TolT>::epsilon()) noexcept {
33 static_assert(std::is_floating_point_v<T>, "Error: Values type must be floating point.");
34 static_assert(std::is_floating_point_v<TolT>, "Error: Tolerance type must be floating point.");
35
36 // Using absolute & relative tolerances for floating points types: http://www.realtimecollisiondetection.net/pubs/Tolerances/
37 // Could be a better idea to use ULPs checking. May be slower though?
38 const T absDiff = std::abs(val1 - val2);
39 return (absDiff <= static_cast<T>(absTol) * std::max(static_cast<T>(1), std::max(std::abs(val1), std::abs(val2))));
40}
