31template <typename T>
32constexpr Quaternion<T> Quaternion<T>::nlerp(const Quaternion& quat, T coeff) const noexcept {
33 assert("Error: The interpolation coefficient must be between 0 & 1." && (coeff >= 0 && coeff <= 1));
34
35 const T currCoeff = 1 - coeff;
36 // A rotation may be represented by two opposite quaternions; should the dot product between those be negative,
37 // one quaternion must be negated. This is made by negating the coefficient
38 const T otherCoeff = (dot(quat) > 0 ? coeff : -coeff);
39
40 return lerp(quat, currCoeff, otherCoeff).normalize();
41}

43template <typename T>
44constexpr Quaternion<T> Quaternion<T>::slerp(const Quaternion& quat, T coeff) const noexcept {
45 assert("Error: The interpolation coefficient must be between 0 & 1." && (coeff >= 0 && coeff <= 1));
46 assert("Error: Quaternions must be normalized for a slerp to be performed." && FloatUtils::areNearlyEqual(computeSquaredNorm(), static_cast<T>(1))
47 && FloatUtils::areNearlyEqual(quat.computeSquaredNorm(), static_cast<T>(1)));
48
49 const T cosAngle = dot(quat);
50
51 T currCoeff {};
52 T otherCoeff {};
53
54 // Checking the angle between the quaternions; if the angle is sufficient, perform an actual spherical interpolation
55 if (std::abs(cosAngle) < static_cast<T>(0.99999)) {
56 const T angle = std::acos(cosAngle);
57 const T invSinAngle = 1 / std::sin(angle);
58
59 currCoeff = std::sin((1 - coeff) * angle) * invSinAngle;
60 otherCoeff = std::sin(coeff * angle) * invSinAngle;
61 } else {
62 // If the angle is really small, a simple linear interpolation is performed
63 currCoeff = 1 - coeff;
64 otherCoeff = coeff;
65 }
66
67 // Negating one quaternion if the angle is negative
68 otherCoeff = (cosAngle > 0 ? otherCoeff : -otherCoeff);
69 return lerp(quat, currCoeff, otherCoeff);
70}

105 const T yz = (2 * m_complexes.y() * m_complexes.z()) * invSqNorm;
106
107 const T xw = (2 * m_complexes.x() * m_real) * invSqNorm;

170template <typename T>
171constexpr Vec3<T> operator*(const Quaternion<T>& quat, const Vec3<T>& vec) noexcept {
172 // Quaternion/vector multiplications are supposed to be made with unit quaternions only, hence the conjugation instead of the inversion
173 // Most likely because of the error accumulation due to floating-point numbers, the norm of a supposedly unit quaternion is almost never very close to 1
174 // That norm is thus not checked here; it may require changing to an actual inversion in the future to avoid further approximation errors
175
176 Quaternion<T> vecQuat(0.f, vec.x(), vec.y(), vec.z());
177 vecQuat = quat * vecQuat * quat.conjugate();
178 return Vec3<T>(vecQuat.x(), vecQuat.y(), vecQuat.z());
179}
