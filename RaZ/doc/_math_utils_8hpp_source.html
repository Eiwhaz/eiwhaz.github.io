155template <typename T>
156void computeOrthonormalBasis(const Vec3<T>& input, Vec3<T>& axis2, Vec3<T>& axis3) {
157 static_assert(std::is_floating_point_v<T>, "Error: Vectors must be of a floating-point type to compute an orthonormal basis from.");
158
159 const T sign = std::copysign(static_cast<T>(1), input.z());
160 const T a = static_cast<T>(-1) / (sign + input.z());
161 const T b = input.x() * input.y() * a;
162
163 axis2 = Vec3<T>(static_cast<T>(1) + sign * input.x() * input.x() * a, sign * b, -sign * input.x());
164 axis3 = Vec3<T>(b, sign + input.y() * input.y() * a, -input.y());
165}
