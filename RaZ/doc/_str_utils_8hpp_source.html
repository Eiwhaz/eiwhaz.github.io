318inline std::vector<std::string> split(std::string text, char delimiter) {
319 trimRight(text);
320
321 std::vector<std::string> parts {};
322
323 while (!text.empty()) {
324 const std::size_t delimPos = text.find_first_of(delimiter);
325
326 if (delimPos > text.size()) {
327 parts.emplace_back(std::move(text));
328 break;
329 }
330
331 parts.emplace_back(text.substr(0, delimPos));
332 trimRight(parts.back());
333
