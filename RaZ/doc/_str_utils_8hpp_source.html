320inline std::vector<std::string> split(std::string text, char delimiter) {
321 trimRight(text);
322
323 std::vector<std::string> parts {};
324
325 while (!text.empty()) {
326 const std::size_t delimPos = text.find_first_of(delimiter);
327
328 if (delimPos > text.size()) {
329 parts.emplace_back(std::move(text));
330 break;
331 }
332
333 parts.emplace_back(text.substr(0, delimPos));
334 trimRight(parts.back());
335
336 text.erase(0, delimPos + 1);
337 trimLeft(text);
338 }
